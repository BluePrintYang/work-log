# 动态规划练习 爬楼梯 (opens new window)

class Solution {
    public int climbStairs(int n) {
        if(n <= 2) return n;
        int n1 = 1;
        int res = 2;
        for(int i = 3; i <= n; i++){
            int tmp = res;
            res += n1;
            n1 = tmp;
        }
        return res;
    }
}

# 使用最小花费爬楼梯 (opens new window)

dp[i]=min(dp[i−1]+cost[i−1],dp[i−2]+cost[i−2])

class Solution {
    public int minCostClimbingStairs(int[] cost) {
        int pre = 0;
        int val = 0;
        for(int i = 2; i <= cost.length; i++){
            int costVal = Math.min(val + cost[i-1], pre + cost[i-2]);
            pre = val;
            val = costVal;
        }
        return val;
    }
}
